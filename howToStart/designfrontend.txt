import { useState } from "react";

export default function App() {
  const [text, setText] = useState("");
  const [result, setResult] = useState(null); // { label, confidence } or error text
  const [loading, setLoading] = useState(false);

  const detectFakeNews = async (inputText = null) => {
    const payloadText = (inputText ?? text).trim();
    if (!payloadText) return;

    setLoading(true);
    setResult(null);

    try {
      const res = await fetch("http://127.0.0.1:5000/predict", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ text: payloadText }),
      });

      const data = await res.json().catch(() => null);

      if (!res.ok) {
        const err = data?.error ?? `HTTP ${res.status}`;
        setResult({ label: `Error: ${err}`, confidence: null });
      } else {
        setResult({ label: data.label ?? data.prediction ?? "Unknown", confidence: data.confidence ?? null });
      }
    } catch (e) {
      setResult({ label: "Backend not connected", confidence: null });
    } finally {
      setLoading(false);
    }
  };

  const onKeyDown = (e) => {
    if (e.key === "Enter" && !e.shiftKey) {
      e.preventDefault();
      detectFakeNews();
    }
  };

  return (
    <div className="min-h-screen bg-white text-slate-900 flex items-center justify-center px-6">
      <div className="w-full max-w-4xl">
        {/* Hero */}
        <header className="text-center pt-24 pb-10">
          <h1 className="text-5xl sm:text-6xl font-extrabold tracking-tight">
            Fake news detection ‚Äî made simple
          </h1>
          <p className="mt-4 text-lg text-slate-600">
            Paste an article or headline and get an instant Real / Fake prediction.
          </p>
          <p className="mt-2 text-sm text-slate-500">
            Uses TF-IDF + your model (served from your Flask API). Try long text for better accuracy.
          </p>
        </header>

        {/* Search bar + detect button (Vercel-style) */}
        <div className="bg-slate-50 border border-slate-100 rounded-2xl p-6 shadow-sm">
          <div className="flex flex-col md:flex-row gap-4 items-stretch">
            <div className="flex-1">
              <label htmlFor="search" className="sr-only">Enter news text</label>
              <textarea
                id="search"
                rows={4}
                value={text}
                onChange={(e) => setText(e.target.value)}
                onKeyDown={onKeyDown}
                placeholder="Paste news, headline, or paragraph here‚Ä¶ (press Enter to detect)"
                className="w-full resize-none px-4 py-3 rounded-xl border border-slate-200 bg-white text-sm focus:outline-none focus:ring-2 focus:ring-slate-300"
              />
              <div className="mt-2 text-xs text-slate-500">
                Tip: longer text usually improves model confidence.
              </div>
            </div>

            <div className="md:w-48 flex-shrink-0 flex items-center">
              <button
                onClick={() => detectFakeNews()}
                disabled={loading}
                className="w-full inline-flex items-center justify-center gap-2 px-4 py-3 bg-slate-900 text-white font-semibold rounded-xl hover:bg-black transition disabled:opacity-60"
              >
                {loading ? (
                  <>
                    <svg className="animate-spin h-4 w-4" viewBox="0 0 24 24" fill="none">
                      <circle cx="12" cy="12" r="10" stroke="currentColor" strokeWidth="4" strokeOpacity="0.25" />
                      <path d="M22 12a10 10 0 0 0-10-10" stroke="currentColor" strokeWidth="4" strokeLinecap="round" />
                    </svg>
                    Detecting...
                  </>
                ) : (
                  "Detect"
                )}
              </button>
            </div>
          </div>
        </div>

        {/* Result card */}
        <div className="mt-6">
          {result ? (
            <div className="rounded-xl p-4 border border-slate-100 bg-white shadow-sm">
              <div className="flex items-center justify-between">
                <div>
                  <div className="text-sm text-slate-500">Prediction</div>
                  <div className="mt-1 text-lg font-semibold">
                    <span className={
                      result.label?.toLowerCase().includes("fake") ? "text-red-600" :
                        result.label?.toLowerCase().includes("real") ? "text-green-600" : "text-slate-700"
                    }>
                      {result.label}
                    </span>
                  </div>
                </div>

                {typeof result.confidence === "number" && (
                  <div className="text-right">
                    <div className="text-sm text-slate-500">Confidence</div>
                    <div className="mt-1 text-base font-medium">{(result.confidence * 100).toFixed(1)}%</div>
                  </div>
                )}
              </div>

              {/* optional progress bar */}
              {typeof result.confidence === "number" && (
                <div className="mt-3 bg-slate-100 rounded-full h-2 overflow-hidden">
                  <div
                    className={`h-2 rounded-full ${result.label?.toLowerCase().includes("fake") ? "bg-red-500" : "bg-green-500"}`}
                    style={{ width: `${Math.min(100, Math.max(0, result.confidence * 100))}%` }}
                  />
                </div>
              )}
            </div>
          ) : (
            <div className="rounded-xl p-4 border border-dashed border-slate-100 text-center text-sm text-slate-500">
              No prediction yet ‚Äî paste text and click Detect.
            </div>
          )}
        </div>

        {/* Footer small CTA */}
        <footer className="mt-10 text-center text-xs text-slate-400 pb-24">
          Powered by your local Flask model ¬∑ Make sure backend runs at <code className="bg-slate-50 px-1 rounded">127.0.0.1:5000</code>
        </footer>
      </div>
    </div>
  );
}
